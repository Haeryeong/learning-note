# vueRouter

싱글 페이지 애플리케이션 구현이 가능

# vueRouter 설치

npm install vue-router

# vueRouter 연결

var MainComponent = {
  template: "<div>main</div>",
};

var router = new VueRouter({
  // 페이지의 라우팅 정보
  routes: [
    {
        //페이지의 url
        path:'/main',
        // 해당 url에서 표시될 컴포넌트
        component: MainComponent
    }, {
        path:,
        component
    }
  ],
});

new Vue({
  router: router,
});

# router-view

router가 가지고 있는 태그

<div id="app">
  <router-view></router-view>
  <!-- LoginComponent 또는 HomeComponent -->
</div>

# router-link

화면에서 특정 링크를 클릭해서 페이지를 이동할 수 있게 해주는 태그

<div>
  <router-link to="/main"></router-link>
</div>
